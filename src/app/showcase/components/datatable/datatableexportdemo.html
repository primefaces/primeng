DataTable - Export DataTable can export its data to CSV format.

Basic

Export selected rows

You can combine selection mode and export. As a result you can use method exportSelectedRowsToCSV() to export selected rows to CSV.

export class DataTableExportDemo implements OnInit {

    cars: Car[];
    
    selectedCars1: Car[];

    constructor(private carService: CarService) { }

    ngOnInit() {
        this.carService.getCarsSmall().then(cars => this.cars = cars);
    }

}
