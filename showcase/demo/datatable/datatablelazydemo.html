DataTable - Lazy Lazy mode is handy to deal with large datasets, instead of loading the entire data, small chunks of data is loaded whenever paging, sorting and filtering happens. onLazyLoad callback is to load the data by getting the state as a parameter.
List of Cars
 

export class DataTablePaginatorDemo implements OnInit {

    cars: Car[];

    cols: Column[];

    constructor(private carService: CarService) { }

    ngOnInit() {
        this.carService.getCarsMedium().then(cars => this.cars = cars);

        this.cols = [
            {field: 'vin', header: 'Vin'},
            {field: 'brand', header: 'Brand'},
            {field: 'year', header: 'Year'},
            {field: 'color', header: 'Color'}
        ];
    }
}



<p-dataTable [value]="cars" [columns]="cols" [rows]="10" [paginator]="true" [pageLinks]="3">
    <header>List of Cars</header>
</p-dataTable>
